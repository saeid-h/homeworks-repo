\begin{homeworkProblem}[Problem \#\arabic{homeworkProblemCounter}]


\begin{enumerate}[label=(\alph*)] % Use to enumerate

\item Create a vector of strings and demonstrate access to elements
\StartLineAt{1} % Start the numeration of the code in 1
\begin{R} %Here copy R code.

x<-c('R','is','a','powerful','open source','statistical','scripting language')
x[1]    #access first element
x[4:6]  #access third thru fifth elements
\end{R}

Output of R console:
\begin{verbatim}
> x[1]    #access first element
[1] "R"
> x[4:6]  #access third thru fifth elements
[1] "powerful"    "open source" "statistical"
\end{verbatim}


\item Use the \texttt{length} function to help insert a new element in to the vector.
\ContinueLineNumber
\begin{R}
#create a new vector "z" from x, but insert "my favorite"
z<-c(x[1:2],"my favorite",x[4:length(x)])   
\end{R}

Output of R console:
\begin{verbatim}
> z  #view the results
[1] "R"                  "is"                 "my favorite"        "powerful"          
[5] "open source"        "statistical"        "scripting language"
\end{verbatim}
\end{enumerate}
\end{homeworkProblem}
